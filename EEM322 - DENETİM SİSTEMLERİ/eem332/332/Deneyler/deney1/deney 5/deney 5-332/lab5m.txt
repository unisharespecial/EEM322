;THE FORM OF AN ASSEMBLY LANGUAGE PROGRAM
;NOTE: USING SIMPLIFIED SEGMENT DEFINITION
		.MODEL SMALL
		.STACK 64
		.DATA
SIGN_DATA 	DB	69,87,96,45,75
ORG 001
HIGHEST	DB	?
		.CODE
MAIN		PROC	FAR
		MOV 	AX,@DATA
		MOV 	DS,AX
		MOV 	CX,5
		MOV 	SI,OFFSET SIGN_DATA
		MOV 	AL,[SI]
BACK:		INC 	SI
		CMP 	AL,[SI]
		JGE 	SEARCH
		MOV 	AL,[SI]
SEARCH:		LOOP 	BACK
		MOV 	HIGHEST,AL
		MOV 	AH,4CH
		INT 	21H
MAIN		ENDP
		END	MAIN
